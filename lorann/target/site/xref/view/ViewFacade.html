31  public class ViewFacade extends JFrame implements IView {
32  	private final JPanel contentPane;

35  	private final Map<String, ImageIcon> images = new HashMap<>();
36  
37  	public ViewFacade() {
38  		super();
39  		setSize(600, 600);
40  		setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
41  		contentPane = new JPanel();
42  		contentPane.setBorder(new EmptyBorder(5, 5, 5, 5));
43  		contentPane.setBackground(Color.BLACK);
44  		contentPane.setFocusable(true);
45  		setContentPane(contentPane);
46  		loadSpriteImages();
47  		setVisible(true);
48  	}

55  	@Override
56  	public void paint(final Graphics g) {
57  		super.paint(g);
58  		try {
59  			if (mapGame != null) {
60  				for (int y = 0; y < GameConstants.MAX_GAME_ROWS; y++) {
61  					for (int x = 0; x < GameConstants.MAX_GAME_COLUMNS; x++) {
62  						final IMapElement element = mapGame.getElement(new Position(x, y));
63  						if (element != null) {
64  							ImageIcon image = null;
65  							if (element instanceof IMonster) {
66  								IMonster monster = (IMonster) element;
67  								switch (monster.getMonsterNumber()) {
68  								case MONSTER1:
69  									image = images.get("monster_1");
70  									break;
71  								case MONSTER2:
72  									image = images.get("monster_2");
73  									break;
74  								case MONSTER3:
75  									image = images.get("monster_3");
76  									break;
77  								case MONSTER4:
78  									image = images.get("monster_4");
79  									break;
80  								}
81  							} else if (element instanceof ILorann) {
82  								drawLorann();
83  								continue;
84  							} else if (element instanceof IPurse) {
85  								image = images.get("purse");
86  							} else if (element instanceof ICball) {
87  								image = images.get("crystal_ball");
88  							} else if (element instanceof IGate) {
89  								image = images.get("gate_open");
90  							} else if (element instanceof IObstacle) {
91  								IObstacle obstacle = (IObstacle) element;
92  								switch (obstacle.getKindOfObstacle()) {
93  								case BALL:
94  									image = images.get("bone");
95  									break;
96  								case H_BONE:
97  									image = images.get("horizontal_bone");
98  									break;
99  								case V_BONE:
100 									image = images.get("vertical_bone");
101 									break;
102 								}
103 							}
104 							g.drawImage(image.getImage(), convX(element.getPosition().getX()),
105 									convY(element.getPosition().getY()), null);
106 						}
107 					}
108 				}
109 			}
110 		} catch (final Exception e) {
111 			e.printStackTrace();
112 		}
113 	}

115 	@Override
116 	public void drawLorann() {
117 		final ILorann lorann = mapGame.getLorann();
118 		getLorannSprite().setLocation(convX(lorann.getPosition().getX()), convY(lorann.getPosition().getY()));
119 	}

121 	@Override
122 	public JPanel getGamePanel() {
123 		return contentPane;
124 	}
125 
126 	/**
127 	 * return an GUI component for Lorann
128 	 * 
129 	 * @return
130 	 */
131 	protected JLabel getLorannSprite() {
132 		if (lorannSprite == null) {
133 			final ImageIcon icon = images.get("lorann_b");
134 			lorannSprite = new JLabel();
135 			lorannSprite.setIcon(icon);
136 			lorannSprite.setSize(GameConstants.SPRITE_WIDTH, GameConstants.SPRITE_HEIGHT);
137 			getGamePanel().add(lorannSprite);
138 		}
139 		return lorannSprite;
140 	}
141 
142 	/**
143 	 * Convert a model X position to x pixel coordinate of view
144 	 * 
145 	 * @param x
146 	 * @return
147 	 */
148 	protected int convX(final int x) {
149 		return x * GameConstants.SPRITE_WIDTH;
150 	}
151 
152 	/**
153 	 * Convert a model Y position to y pixel coordinate of view
154 	 * 
155 	 * @param y
156 	 * @return
157 	 */
158 	protected int convY(final int y) {
159 		return (GameConstants.MAX_GAME_ROWS - y - 1) * GameConstants.SPRITE_HEIGHT;
160 	}
